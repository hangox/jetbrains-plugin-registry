# 测试方案

## 1. 测试分层

```
┌─────────────────────────────────────────────────┐
│  功能测试 (Gradle TestKit)                        │
│  模拟完整 Gradle 构建 + Task 执行                  │
│  验证 DSL → Task → HTTP 全链路                     │
├─────────────────────────────────────────────────┤
│  集成测试                                         │
│  Mock HTTP Server 验证请求格式                     │
│  验证 Multipart 构建、响应处理                      │
├─────────────────────────────────────────────────┤
│  单元测试                                         │
│  JSON 解析、URL 构建、文件查找逻辑                   │
└─────────────────────────────────────────────────┘
```

## 2. 单元测试

### 2.1 URL 构建

```kotlin
// UploadPluginTaskTest.kt
class UploadPluginTaskTest {

    @Test
    fun `buildUploadUrl without force`() {
        val url = buildUploadUrl("https://plugins.example.com", false)
        assertEquals("https://plugins.example.com/api/plugins?force=false", url)
    }

    @Test
    fun `buildUploadUrl with force`() {
        val url = buildUploadUrl("https://plugins.example.com/", true)
        assertEquals("https://plugins.example.com/api/plugins?force=true", url)
    }

    @Test
    fun `buildUploadUrl trims trailing slash`() {
        val url = buildUploadUrl("https://plugins.example.com///", false)
        assertEquals("https://plugins.example.com/api/plugins?force=false", url)
    }
}
```

## 3. 集成测试（Mock HTTP Server）

使用 JDK 内置的 `com.sun.net.httpserver.HttpServer` 作为 Mock Server：

```kotlin
// UploadPluginIntegrationTest.kt
import com.sun.net.httpserver.HttpServer
import java.net.InetSocketAddress

class UploadPluginIntegrationTest {

    private lateinit var mockServer: HttpServer
    private var lastRequestBody: ByteArray? = null
    private var lastRequestHeaders: Map<String, List<String>>? = null
    private var responseCode = 201
    private var responseBody = """{"id":"test","version":"1.0.0","sinceBuild":"222"}"""

    @BeforeEach
    fun setUp() {
        mockServer = HttpServer.create(InetSocketAddress(0), 0)
        mockServer.createContext("/api/plugins") { exchange ->
            // 记录请求
            lastRequestBody = exchange.requestBody.readAllBytes()
            lastRequestHeaders = exchange.requestHeaders

            // 返回预设响应
            val response = responseBody.toByteArray()
            exchange.sendResponseHeaders(responseCode, response.size.toLong())
            exchange.responseBody.use { it.write(response) }
        }
        mockServer.start()
    }

    @AfterEach
    fun tearDown() {
        mockServer.stop(0)
    }

    @Test
    fun `upload sends correct multipart request`() {
        val port = mockServer.address.port
        val zipFile = createTestZip()

        // 执行上传
        sendRequest("http://localhost:$port", "test-token", zipFile, false)

        // 验证 Authorization 头
        val authHeader = lastRequestHeaders!!["Authorization"]!!.first()
        assertEquals("Bearer test-token", authHeader)

        // 验证 Content-Type 包含 boundary
        val contentType = lastRequestHeaders!!["Content-type"]!!.first()
        assertTrue(contentType.startsWith("multipart/form-data; boundary="))

        // 验证 body 包含文件名
        val bodyStr = String(lastRequestBody!!)
        assertTrue(bodyStr.contains("filename=\"test-plugin.zip\""))
    }

    @Test
    fun `upload handles 409 conflict`() {
        responseCode = 409
        responseBody = """{"error":"Version already exists"}"""

        val port = mockServer.address.port
        val zipFile = createTestZip()

        val exception = assertThrows<RuntimeException> {
            sendRequest("http://localhost:$port", "test-token", zipFile, false)
        }
        assertTrue(exception.message!!.contains("already exists"))
        assertTrue(exception.message!!.contains("--force"))
    }

    @Test
    fun `upload handles 401 unauthorized`() {
        responseCode = 401
        responseBody = """{"error":"Unauthorized"}"""

        val port = mockServer.address.port
        val zipFile = createTestZip()

        val exception = assertThrows<RuntimeException> {
            sendRequest("http://localhost:$port", "bad-token", zipFile, false)
        }
        assertTrue(exception.message!!.contains("Authentication failed"))
    }

    @Test
    fun `upload sends force=true in URL`() {
        val port = mockServer.address.port
        val zipFile = createTestZip()

        mockServer.removeContext("/api/plugins")
        var requestUri: String? = null
        mockServer.createContext("/api/plugins") { exchange ->
            requestUri = exchange.requestURI.toString()
            val response = responseBody.toByteArray()
            exchange.sendResponseHeaders(201, response.size.toLong())
            exchange.responseBody.use { it.write(response) }
        }

        sendRequest("http://localhost:$port", "test-token", zipFile, true)

        assertEquals("/api/plugins?force=true", requestUri)
    }

    private fun createTestZip(): java.io.File {
        val tmp = Files.createTempFile("test-plugin", ".zip").toFile()
        tmp.writeBytes(byteArrayOf(0x50, 0x4B, 0x03, 0x04))  // ZIP magic bytes
        return tmp
    }
}
```

## 4. 功能测试（Gradle TestKit）

Gradle TestKit 会启动真实的 Gradle 进程执行 Task，验证完整链路：

```kotlin
// UploadPluginFunctionalTest.kt
import org.gradle.testkit.runner.GradleRunner
import org.gradle.testkit.runner.TaskOutcome

class UploadPluginFunctionalTest {

    @TempDir
    lateinit var projectDir: java.io.File

    private lateinit var mockServer: HttpServer

    @BeforeEach
    fun setUp() {
        // 启动 Mock Server
        mockServer = HttpServer.create(InetSocketAddress(0), 0)
        mockServer.createContext("/api/plugins") { exchange ->
            val response = """{"id":"test","version":"1.0.0","sinceBuild":"222"}"""
                .toByteArray()
            exchange.sendResponseHeaders(201, response.size.toLong())
            exchange.responseBody.use { it.write(response) }
        }
        mockServer.start()

        // 创建测试项目结构
        val port = mockServer.address.port

        // settings.gradle.kts
        java.io.File(projectDir, "settings.gradle.kts").writeText("""
            rootProject.name = "test-project"
        """.trimIndent())

        // build.gradle.kts（不依赖 IntelliJ Gradle Plugin，手动模拟 buildPlugin）
        java.io.File(projectDir, "build.gradle.kts").writeText("""
            plugins {
                id("com.example.private-plugin-registry")
            }

            privateRegistry {
                url = "http://localhost:$port"
                token = "test-token"
            }

            // 模拟 buildPlugin task
            tasks.register("buildPlugin") {
                val distDir = layout.buildDirectory.dir("distributions")
                outputs.dir(distDir)
                doLast {
                    val dir = distDir.get().asFile
                    dir.mkdirs()
                    java.io.File(dir, "test-plugin-1.0.0.zip").writeBytes(
                        byteArrayOf(0x50, 0x4B, 0x03, 0x04, 0x00, 0x00)
                    )
                }
            }
        """.trimIndent())
    }

    @AfterEach
    fun tearDown() {
        mockServer.stop(0)
    }

    @Test
    fun `uploadPlugin task succeeds with valid config`() {
        val result = GradleRunner.create()
            .withProjectDir(projectDir)
            .withPluginClasspath()
            .withArguments("uploadPlugin")
            .build()

        assertEquals(TaskOutcome.SUCCESS, result.task(":uploadPlugin")?.outcome)
        assertTrue(result.output.contains("Upload successful"))
    }

    @Test
    fun `uploadPlugin fails without token`() {
        // 重写 build.gradle.kts 去掉 token
        java.io.File(projectDir, "build.gradle.kts").writeText("""
            plugins {
                id("com.example.private-plugin-registry")
            }
            privateRegistry {
                url = "http://localhost:${mockServer.address.port}"
            }
            tasks.register("buildPlugin") {
                doLast {
                    val dir = layout.buildDirectory.dir("distributions").get().asFile
                    dir.mkdirs()
                    java.io.File(dir, "test.zip").writeBytes(byteArrayOf(0x50, 0x4B))
                }
            }
        """.trimIndent())

        val result = GradleRunner.create()
            .withProjectDir(projectDir)
            .withPluginClasspath()
            .withArguments("uploadPlugin")
            .buildAndFail()

        assertEquals(TaskOutcome.FAILED, result.task(":uploadPlugin")?.outcome)
        assertTrue(result.output.contains("token is required"))
    }

    @Test
    fun `uploadPlugin task applies correct group and description`() {
        val result = GradleRunner.create()
            .withProjectDir(projectDir)
            .withPluginClasspath()
            .withArguments("tasks", "--group", "publishing")
            .build()

        assertTrue(result.output.contains("uploadPlugin"))
        assertTrue(result.output.contains("Build and upload plugin to private registry"))
    }

    @Test
    fun `force flag works from command line`() {
        var receivedUri: String? = null
        mockServer.removeContext("/api/plugins")
        mockServer.createContext("/api/plugins") { exchange ->
            receivedUri = exchange.requestURI.toString()
            val resp = """{"id":"test","version":"1.0.0","sinceBuild":"222"}""".toByteArray()
            exchange.sendResponseHeaders(201, resp.size.toLong())
            exchange.responseBody.use { it.write(resp) }
        }

        GradleRunner.create()
            .withProjectDir(projectDir)
            .withPluginClasspath()
            .withArguments("uploadPlugin", "--force")
            .build()

        assertEquals("/api/plugins?force=true", receivedUri)
    }
}
```

## 5. 测试依赖配置

```kotlin
// build.gradle.kts
dependencies {
    testImplementation(gradleTestKit())
    testImplementation("org.junit.jupiter:junit-jupiter:5.11.0")
}

// 可选：分离功能测试 source set
sourceSets {
    create("functionalTest") {
        compileClasspath += sourceSets.main.get().output
        runtimeClasspath += sourceSets.main.get().output
    }
}

val functionalTestImplementation by configurations.getting {
    extendsFrom(configurations.testImplementation.get())
}

val functionalTestRuntimeOnly by configurations.getting {
    extendsFrom(configurations.testRuntimeOnly.get())
}

tasks.register<Test>("functionalTest") {
    testClassesDirs = sourceSets["functionalTest"].output.classesDirs
    classpath = sourceSets["functionalTest"].runtimeClasspath
    useJUnitPlatform()
}
```

```bash

# 功能测试
./gradlew functionalTest
```
